pub enum Expression<'a> {
    Null,
    Number(Number<'a>),
    Boolean(bool),
    String(Cow<'a, str>),
    Tuple(Tuple<'a>),
    Object(Object<'a>),
}

An Expression

Expression = (
    ExprTerm |
    Operation |  # Not supported
    Conditional # Not supported
);

ExprTerm = (
    LiteralValue |
    CollectionValue |
    TemplateExpr |
    VariableExpr |
    FunctionCall |
    ForExpr |
    ExprTerm Index |
    ExprTerm GetAttr |
    ExprTerm Splat |
    "(" Expression ")"
);

LiteralValue = (
  NumericLit |
  "true" |
  "false" |
  "null"
);
  • Numeric literals represent values of type number.

Variants

Null

A null HCL expression, expressed literally

Number(Number<'a>)

An arbitrary precision number

Boolean(bool)

A boolean value, expressed as the literals true or false

String(Cow<'a, str>)

A HCL string

Tuple(Tuple<'a>)

A HCL tuple (list)

Object(Object<'a>)

A HCL object (map)
